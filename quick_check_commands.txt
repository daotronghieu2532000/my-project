# CÁC LỆNH KIỂM TRA NHANH - CHẠY TỪNG LỆNH MỘT

# 1. Kiểm tra PM2 processes
pm2 list

# 2. Kiểm tra port 3000 đang được dùng bởi process nào
lsof -i :3000

# 3. Kiểm tra tất cả Node.js processes và port của chúng
ps aux | grep node | grep -v grep
netstat -tulpn | grep node

# 4. Kiểm tra Nginx HTTPS config (quan trọng nhất)
cat /etc/nginx/config-https/chat.socdo.vn-https.conf

# 5. Test kết nối Socket.IO
curl -I https://chat.socdo.vn/socket.io/
